    <script>

        async function fetchVacancies() {

            const rssInput = 'https://mycareer.bathspa.ac.uk/content/rss/MyCareerPTJobs/e7ff9b8c15769a944e719b4033dea757bf1a4a8a';

            const track = document.getElementById('vacancies-track');

            

            if(!rssInput) {

                track.innerHTML = '<div class="loading">Enter your Vacancies RSS URL to display jobs.</div>';

                return;

            }


            const proxyUrl = 'https://api.rss2json.com/v1/api.json?rss_url=' + encodeURIComponent(rssInput);


            try {

                const response = await fetch(proxyUrl);

                const data = await response.json();

                if (data && data.items && data.items.length > 0) {

                    renderVacancies(data.items, track);

                } else {

                    track.innerHTML = '<div class="loading">No current vacancies found.</div>';

                }

            } catch (error) {

                console.error("Error fetching RSS:", error);

                track.innerHTML = '<div class="loading" style="color: #ef4444;">Unable to load vacancies. Check feed URL.</div>';

            }

        }


        function renderVacancies(items, track) {

            track.innerHTML = ''; 


            items.forEach(item => {

                const dDoc = new DOMParser().parseFromString(item.description, "text/html");

                

                let employer = "Employer";

                let oppType = "Opportunity";

                let deadline = "Open";

                let details = "";


                // TargetConnect Specific HTML Parsing

                const labels = dDoc.querySelectorAll('.feed-entry-label');

                labels.forEach(label => {

                    const txt = label.textContent.toLowerCase();

                    const parent = label.parentElement;

                    const valNode = parent.querySelector('.feed-entry-value');


                    if (txt.includes('employer name')) {

                        employer = valNode ? valNode.textContent.trim() : "Unknown Employer";

                    }

                    if (txt.includes('type of opportunity')) {

                        oppType = valNode ? valNode.textContent.trim() : "Opportunity";

                    }

                    if (txt.includes('closing date')) {

                        deadline = valNode ? valNode.textContent.trim() : "Ongoing";

                        // Simplify time output if it includes it

                        if(deadline.includes(' at ')) deadline = deadline.split(' at ')[0];

                    }

                    if (txt.includes('details')) {

                        if (valNode) {

                            details = valNode.textContent.trim();

                        } else {

                            label.remove(); 

                            details = parent.textContent.trim();

                        }

                        details = details.replace(/^:\s*/, '').trim();

                    }

                });


                if (details.startsWith(':')) details = details.substring(1).trim();

                if (!details) details = "Click to view full job description and application details.";


                const card = document.createElement('a');

                card.href = item.link;

                card.target = "_blank";

                card.className = 'vacancy-card';


                const initial = employer.charAt(0).toUpperCase();


                card.innerHTML = `

                    <div class="card-header">

                        <div class="employer-icon">${initial}</div>

                        <div class="header-text">

                            <div class="employer-name">${employer}</div>

                            <h3 class="vacancy-title">${item.title}</h3>

                        </div>

                    </div>

                    <div class="pill-container">

                        <span class="pill pill-type">💼 ${oppType}</span>

                        <span class="pill pill-deadline">⏳ Closes ${deadline}</span>

                    </div>

                    <p class="vacancy-desc">${details}</p>

                `;


                track.appendChild(card);

            });

        }

        fetchVacancies();

    </script>
